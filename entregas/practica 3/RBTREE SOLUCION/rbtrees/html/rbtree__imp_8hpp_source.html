1260 template <class T>
1261 bool RBTree<T>::check_parent_chains()
1262 {
1263  if (!is_empty())
1264  {
1265  std::function<void(typename RBTNode<T>::Ref, std::vector<T>)> go_down;
1266  go_down = [&go_down](typename RBTNode<T>::Ref node, std::vector<T> branch) -> void
1267  {
1268  if (node->left() != nullptr || node->right() != nullptr)
1269  {
1270  branch.push_back(node->item());
1271  if (node->left())
1272  // go down by the left
1273  go_down(node->left(), branch);
1274  if (node->right())
1275  // go down by the right
1276  go_down(node->right(), branch);
1277  }
1278  else
1279  {
1280  // The node is a leaf node, so check the branch
1281  // to the tree root node.
1282  typename RBTNode<T>::Ref parent = node->parent();
1283  int idx = static_cast<int>(branch.size()) - 1;
1284  while (parent && idx >= 0)
1285  {
1286  assert(parent->item() == branch[idx]);
1287  --idx;
1288  parent = parent->parent();
1289  }
1290  assert(idx == -1 && parent == nullptr);
1291  }
1292  };
1293  std::vector<T> branch;
1294  go_down(root_node(), branch);
1295  }
1296  return true;
1297 }

1299 template <class T>
1300 std::tuple<size_t, size_t>
1301 compute_min_max_branch_length(typename RBTree<T>::Ref tree)
1302 {
1303  assert(tree != nullptr);
1304  size_t min_path_l = 0;
1305  size_t max_path_l = 0;
1306  // TODO
1307  // Hint: you can use a lambda function with prototype:
1308  // std::function<void(typename RBTree<T>::Ref subt, size_t & min, size_t & max, size_t curr_l)>
1309  // to recursive go down through the tree.
1310  // Each new recursion increases the level in the tree. When a subtree is empty
1311  // you have achieved a leaf of a branch and the current level is the
1312  // branch length. Update the min/max values according before returning from
1313  // this recursion.
1314  // See: https://stackoverflow.com/questions/2067988/recursive-lambda-functions-in-c11
1315  // to study a similar case.
1316

1318  go_down = [&go_down](typename RBTree<T>::Ref subtree, size_t &min_path_l,
1319  size_t &max_path_l, size_t current_level) -> void
1320  {
1321  assert(!subtree->is_empty());
1322  auto left = subtree->left();
1323  auto right = subtree->right();
1324  if (left->is_empty() && right->is_empty())
1325  {
1326  min_path_l = std::min(min_path_l, current_level);
1327  max_path_l = std::max(max_path_l, current_level);
1328  }
1329  else
1330  {
1331  if (!left->is_empty())
1332  go_down(left, min_path_l, max_path_l, current_level + 1);
1333  if (!right->is_empty())
1334  go_down(right, min_path_l, max_path_l, current_level + 1);
1335  }
1336  };
1337  if (!tree->is_empty())
1338  {
1339  min_path_l = std::numeric_limits<size_t>::max();
1340  max_path_l = 0;
1341  go_down(tree, min_path_l, max_path_l, 0);
1342  }
1343  //
1344 
1345  return std::make_tuple(min_path_l, max_path_l);
1346 }
